cmake_minimum_required(VERSION 4.2.0)
project(
    SimpleGame
    LANGUAGES CXX 
)
# 设置C语言标准
set(CMAKE_C_STANDARD 11)

set(SDL2_PATH "C:/deps/SDL2")
include_directories("${SDL2_PATH}/include")
link_directories("${SDL2_PATH}/lib/x64")

include_directories("include")
add_executable(${PROJECT_NAME} src/main.cpp src/Entity.cpp src/BackGround.cpp)



target_link_libraries(${PROJECT_NAME} SDL2main SDL2)

set(CMAKE_EXPORT_COMPILE_COMMANDS ON)

# ---------------------------------------------------------
# SDL2 DLL 自动处理逻辑
# ---------------------------------------------------------

# 定义 DLL 的源路径 (根据你的实际路径修改)
set(SDL2_DLL_PATH "${SDL2_PATH}/lib/x64/SDL2.dll")

# 添加一个构建后事件
add_custom_command(TARGET ${PROJECT_NAME} POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_if_different
        "${SDL2_DLL_PATH}"
        "$<TARGET_FILE_DIR:${PROJECT_NAME}>"
    COMMENT "正在将 SDL2.dll 复制到输出目录..."
)
